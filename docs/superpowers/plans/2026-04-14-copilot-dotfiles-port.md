# Copilot Dotfiles Port Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port the claude-dotfiles configuration system to Copilot CLI via a one-way sync transform, with plugin management for both platforms.

**Architecture:** A Node.js sync script in claude-dotfiles reads command markdown files + a mapping config, applies tool-name substitutions and format transforms, and writes the full copilot-dotfiles tree. An install script in copilot-dotfiles symlinks the output into `~/.copilot/` and installs plugins.

**Tech Stack:** Node.js (ES modules), YAML (js-yaml), Bash (install scripts)

### Task 4: Create the sync script (copilot-sync.mjs)

**Files:**
- Create: `~/projects/claude-dotfiles/sync/copilot-sync.mjs`

This is the core transform engine. It reads the mapping config and commands, then generates the entire copilot-dotfiles tree.

- [ ] **Step 1: Write the sync script**

Create `sync/copilot-sync.mjs` with the following structure:

```javascript
#!/usr/bin/env node

/**
 * copilot-sync.mjs — One-way transform from claude-dotfiles to copilot-dotfiles.
 *
 * Reads commands/*.md + CLAUDE.md + mapping config + plugins manifest,
 * generates the full copilot-dotfiles tree (skills, agents, config, install script).
 *
 * Usage: node sync/copilot-sync.mjs [--dry-run]
 */

import { readFileSync, writeFileSync, mkdirSync, cpSync, rmSync, existsSync, readdirSync, statSync } from 'fs';
import { resolve, join, dirname, basename } from 'path';
import { homedir } from 'os';
import { load as yamlLoad } from 'js-yaml';

const CLAUDE_ROOT = resolve(dirname(new URL(import.meta.url).pathname), '..');
const DRY_RUN = process.argv.includes('--dry-run');

// ─── Load configs ───────────────────────────────────────────────────────────

function loadYaml(path) {
  return yamlLoad(readFileSync(path, 'utf8'));
}

const mapping = loadYaml(join(CLAUDE_ROOT, 'sync/copilot-mapping.yaml'));
const plugins = loadYaml(join(CLAUDE_ROOT, 'plugins.yaml'));
const outputDir = mapping.output_dir.replace('~', homedir());

// ─── Helpers ────────────────────────────────────────────────────────────────

function ensureDir(dir) {
  if (!DRY_RUN) mkdirSync(dir, { recursive: true });
}

function writeOut(path, content) {
  if (DRY_RUN) {
    console.log(`  [dry-run] would write: ${path}`);
    return;
  }
  ensureDir(dirname(path));
  writeFileSync(path, content, 'utf8');
}

function copyDir(src, dest) {
  if (DRY_RUN) {
    console.log(`  [dry-run] would copy: ${src} → ${dest}`);
    return;
  }
  ensureDir(dirname(dest));
  cpSync(src, dest, { recursive: true });
}

/** Apply tool_map + text_substitutions to a markdown body. */
function applySubstitutions(text) {
  let result = text;

  // Tool name substitutions — match as whole words in prose context
  for (const [claude, copilot] of Object.entries(mapping.tool_map)) {
    // Replace backtick-wrapped tool names: `Read` → `view`
    result = result.replaceAll(`\`${claude}\``, `\`${copilot}\``);
    // Replace tool names in "Use X/Y/Z directly" patterns
    result = result.replaceAll(`/${claude}/`, `/${copilot}/`);
  }

  // Text substitutions (exact string match)
  for (const [from, to] of Object.entries(mapping.text_substitutions)) {
    result = result.replaceAll(from, to);
  }

  return result;
}

/** Extract the first paragraph after the H1 title. */
function extractDescription(markdown) {
  const lines = markdown.split('\n');
  let foundTitle = false;
  let desc = [];
  for (const line of lines) {
    if (line.startsWith('# ')) { foundTitle = true; continue; }
    if (!foundTitle) continue;
    if (line.trim() === '' && desc.length === 0) continue; // skip blank after title
    if (line.trim() === '' && desc.length > 0) break; // end of first paragraph
    if (line.startsWith('##')) break; // hit next section
    desc.push(line.trim());
  }
  return desc.join(' ') || 'No description';
}

// ─── Transform: Command → SKILL.md ─────────────────────────────────────────

function transformCommand(name, config) {
  const cmdPath = join(CLAUDE_ROOT, 'commands', `${name}.md`);
  if (!existsSync(cmdPath)) {
    console.warn(`  WARN: command not found: ${cmdPath}`);
    return null;
  }

  const raw = readFileSync(cmdPath, 'utf8');
  const description = extractDescription(raw);
  const body = applySubstitutions(raw);

  const frontmatter = [
    '---',
    `name: ${name}`,
    `description: ${description}`,
    'allowed-tools:',
    ...config.allowed_tools.map(t => `  - ${t}`),
    '---',
    '',
  ].join('\n');

  return frontmatter + body;
}

// ─── Transform: Alias → .agent.md ──────────────────────────────────────────

function generateAgent(alias, skillName, description) {
  return [
    '---',
    `name: ${alias}`,
    `description: Alias for ${skillName} — ${description}`,
    '---',
    '',
    `Run the ${skillName} skill. Pass through any arguments the user provided.`,
    '',
  ].join('\n');
}

// ─── Transform: CLAUDE.md → AGENTS.md ───────────────────────────────────────

function transformClaudeMdToAgents() {
  const raw = readFileSync(join(CLAUDE_ROOT, 'CLAUDE.md'), 'utf8');
  let content = applySubstitutions(raw);

  // Replace the title
  content = content.replace(
    '# Universal Director - Claude Code',
    '# Universal Director'
  );

  // Replace the subtitle
  content = content.replace(
    'This is my global Claude Code configuration.',
    'This is my global configuration.'
  );

  // Remove the footer about symlinks
  content = content.replace(
    /\n---\n\n\*This file is managed.*$/s,
    '\n---\n\n*This file is generated by copilot-sync from claude-dotfiles. Do not edit directly.*\n'
  );

  return content;
}

// ─── Transform: CLAUDE.md → copilot-instructions.md ─────────────────────────

function transformClaudeMdToUserLevel() {
  const raw = readFileSync(join(CLAUDE_ROOT, 'CLAUDE.md'), 'utf8');
  const sections = mapping.user_level_sections;

  // Parse markdown into sections by H2
  const lines = raw.split('\n');
  let currentSection = null;
  const sectionMap = {};
  const headerLine = [];

  for (const line of lines) {
    if (line.startsWith('# ') && !line.startsWith('## ')) {
      // Title — keep it
      headerLine.push(line);
      continue;
    }
    if (line.startsWith('## ')) {
      currentSection = line.replace('## ', '').trim();
      sectionMap[currentSection] = [line];
      continue;
    }
    if (currentSection) {
      sectionMap[currentSection].push(line);
    }
  }

  // Build output with only user-level sections
  const parts = [
    '# User Instructions',
    '',
    '> Personal development preferences and style. Applies to all projects.',
    '',
  ];

  for (const sectionName of sections) {
    if (sectionMap[sectionName]) {
      parts.push(...sectionMap[sectionName]);
      parts.push('');
    }
  }

  parts.push('---');
  parts.push('');
  parts.push('*Generated by copilot-sync from claude-dotfiles. Do not edit directly.*');
  parts.push('');

  return applySubstitutions(parts.join('\n'));
}

// ─── Generate: config.json ──────────────────────────────────────────────────

function generateConfig() {
  const config = {
    model: 'claude-sonnet-4-6',
    statusLine: `node ${outputDir}/lib/statusline.cjs`,
    includeCoAuthoredBy: true,
    hooks: {
      sessionStart: [],
      preToolUse: [],
      postToolUse: [],
    },
  };
  return JSON.stringify(config, null, 2) + '\n';
}

// ─── Generate: mcp-config.json ──────────────────────────────────────────────

function generateMcpConfig() {
  const servers = {};
  for (const [name, cfg] of Object.entries(plugins.mcp_servers || {})) {
    const args = cfg.args.map(a => {
      if (cfg.relative) return resolve(outputDir, a);
      return a;
    });
    servers[name] = {
      type: 'stdio',
      command: cfg.command,
      args,
    };
  }
  return JSON.stringify({ mcpServers: servers }, null, 2) + '\n';
}

// ─── Generate: .copilot-generated marker ────────────────────────────────────

function generateMarker() {
  return [
    '# This directory is generated by copilot-sync from claude-dotfiles.',
    '# Do not edit files here directly — changes will be overwritten.',
    '#',
    `# Source: ${CLAUDE_ROOT}`,
    `# Generated: ${new Date().toISOString()}`,
    `# Sync command: just sync-copilot (in claude-dotfiles)`,
    '',
  ].join('\n');
}

// ─── Generate: install.sh ───────────────────────────────────────────────────

function generateInstallScript() {
  const templatePath = join(CLAUDE_ROOT, 'sync/templates/install.sh.template');
  if (!existsSync(templatePath)) {
    console.warn('  WARN: install.sh.template not found, skipping install script generation');
    return null;
  }
  let template = readFileSync(templatePath, 'utf8');

  // Inject plugin install commands from manifest
  const pluginCommands = [];
  for (const [name, cfg] of Object.entries(plugins.plugins || {})) {
    if (cfg.copilot) {
      pluginCommands.push(`# ${name}`);
      if (cfg.copilot.marketplace_source) {
        pluginCommands.push(`copilot plugin marketplace add ${cfg.copilot.marketplace_source} 2>/dev/null || true`);
      }
      pluginCommands.push(`copilot plugin install ${cfg.copilot.marketplace}/${name} 2>/dev/null || log_warn "${name} plugin install failed — install manually"`);
    }
    if (cfg.type === 'cli') {
      pluginCommands.push(`# ${name} (CLI tool)`);
      pluginCommands.push(`${cfg.install} || log_warn "${name} install failed — install manually"`);
    }
  }

  template = template.replace('{{PLUGIN_INSTALL_COMMANDS}}', pluginCommands.join('\n    '));
  template = template.replace('{{DOTFILES_DIR}}', outputDir);

  return template;
}

// ─── Clean output directory ─────────────────────────────────────────────────

function cleanOutput() {
  if (!existsSync(outputDir)) {
    ensureDir(outputDir);
    return;
  }

  // Preserve .git, docs/, .copilot-generated
  const preserve = new Set(['.git', 'docs', '.copilot-generated']);
  for (const entry of readdirSync(outputDir)) {
    if (preserve.has(entry)) continue;
    const fullPath = join(outputDir, entry);
    if (DRY_RUN) {
      console.log(`  [dry-run] would remove: ${fullPath}`);
    } else {
      rmSync(fullPath, { recursive: true, force: true });
    }
  }
}

// ─── Main ───────────────────────────────────────────────────────────────────

function main() {
  console.log(`Copilot Sync: ${CLAUDE_ROOT} → ${outputDir}`);
  if (DRY_RUN) console.log('  (dry-run mode)\n');

  // Step 1: Clean
  console.log('Cleaning output directory...');
  cleanOutput();

  // Step 2: Transform commands → skills
  let skillCount = 0;
  let agentCount = 0;
  console.log('Transforming commands → skills...');

  for (const [name, config] of Object.entries(mapping.commands)) {
    const skill = transformCommand(name, config);
    if (skill) {
      writeOut(join(outputDir, 'skills', name, 'SKILL.md'), skill);
      skillCount++;

      // Generate alias agent if configured
      if (config.alias) {
        const desc = extractDescription(
          readFileSync(join(CLAUDE_ROOT, 'commands', `${name}.md`), 'utf8')
        );
        const agent = generateAgent(config.alias, name, desc);
        writeOut(join(outputDir, 'agents', `${config.alias}.agent.md`), agent);
        agentCount++;
      }
    }
  }

  // Step 3: Transform CLAUDE.md
  console.log('Transforming CLAUDE.md...');
  writeOut(join(outputDir, 'AGENTS.md'), transformClaudeMdToAgents());
  writeOut(join(outputDir, 'copilot-instructions.md'), transformClaudeMdToUserLevel());

  // Step 4: Generate configs
  console.log('Generating configs...');
  writeOut(join(outputDir, 'config.json'), generateConfig());
  writeOut(join(outputDir, 'mcp-config.json'), generateMcpConfig());

  // Step 5: Copy shared assets
  console.log('Copying shared assets...');
  for (const asset of mapping.shared) {
    const src = join(CLAUDE_ROOT, asset);
    const dest = join(outputDir, asset);
    if (existsSync(src)) {
      copyDir(src, dest);
    } else {
      console.warn(`  WARN: shared asset not found: ${src}`);
    }
  }

  // Step 6: Generate install script
  console.log('Generating install script...');
  const installScript = generateInstallScript();
  if (installScript) {
    writeOut(join(outputDir, 'install.sh'), installScript);
  }

  // Step 7: Write marker
  writeOut(join(outputDir, '.copilot-generated'), generateMarker());

  // Report
  console.log(`\nSync complete:`);
  console.log(`  Skills:  ${skillCount}`);
  console.log(`  Agents:  ${agentCount}`);
  console.log(`  Shared:  ${mapping.shared.length} assets`);
  console.log(`  Output:  ${outputDir}`);
}

main();
```

```bash
#!/usr/bin/env bash
set -euo pipefail

# ─── Copilot Dotfiles Installer ─────────────────────────────────────────────
# Doctor mode: checks → reports → offers fixes.
# Generated by copilot-sync from claude-dotfiles.
#
# Usage:
#   ./install.sh              # interactive doctor mode
#   ./install.sh --dry-run    # show what would change
#   ./install.sh --force      # auto-fix without prompting
#   ./install.sh --uninstall  # remove symlinks only

DOTFILES_DIR="$(cd "$(dirname "$0")" && pwd)"
COPILOT_DIR="$HOME/.copilot"
MCP_CONFIG="$COPILOT_DIR/mcp-config.json"

# ─── Options ─────────────────────────────────────────────────────────────────
DRY_RUN=false
FORCE=false
UNINSTALL=false
for arg in "$@"; do
  case "$arg" in
    --dry-run)   DRY_RUN=true ;;
    --force)     FORCE=true ;;
    --uninstall) UNINSTALL=true ;;
  esac
done

# ─── Logging ─────────────────────────────────────────────────────────────────
RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[0;33m'; CYAN='\033[0;36m'; NC='\033[0m'
OK="${GREEN}OK${NC}"; WARN="${YELLOW}WARN${NC}"; FAIL="${RED}FAIL${NC}"; FIX="${CYAN}FIX${NC}"
CHECKS=0; PASSED=0; FIXED=0; FAILED=0

log_ok()   { ((CHECKS++)); ((PASSED++)); echo -e "  [$OK]   $1"; }
log_warn() { ((CHECKS++)); echo -e "  [$WARN] $1"; }
log_fail() { ((CHECKS++)); ((FAILED++)); echo -e "  [$FAIL] $1"; }
log_fix()  { ((FIXED++)); echo -e "  [$FIX]  $1"; }

prompt_fix() {
  if $FORCE; then return 0; fi
  if $DRY_RUN; then echo "  [dry-run] would fix: $1"; return 1; fi
  read -rp "  Fix: $1? [y/N] " answer
  [[ "$answer" =~ ^[Yy] ]]
}

# ─── Uninstall ───────────────────────────────────────────────────────────────
if $UNINSTALL; then
  echo "Removing copilot-dotfiles symlinks..."
  rm -f "$COPILOT_DIR/copilot-instructions.md"
  rm -f "$COPILOT_DIR/skills"
  rm -f "$COPILOT_DIR/agents"
  echo "Done. Config and MCP registration left intact."
  exit 0
fi

# ─── Phase 1: Prerequisites ─────────────────────────────────────────────────
echo ""
echo "Phase 1: Prerequisites"
echo "─────────────────────"

for cmd in node git copilot just gh; do
  if command -v "$cmd" &>/dev/null; then
    log_ok "$cmd found: $(command -v "$cmd")"
  else
    log_fail "$cmd not found"
    case "$cmd" in
      node)    echo "         Install: https://nodejs.org/" ;;
      copilot) echo "         Install: npm install -g @github/copilot" ;;
      just)    echo "         Install: cargo install just OR https://just.systems/man/en/installation.html" ;;
      gh)      echo "         Install: https://cli.github.com/" ;;
    esac
  fi
done

# ─── Phase 2: Directory Structure ───────────────────────────────────────────
echo ""
echo "Phase 2: Directory Structure"
echo "────────────────────────────"

mkdir -p "$COPILOT_DIR"
log_ok "~/.copilot/ exists"

# ─── Phase 3: Symlinks ──────────────────────────────────────────────────────
echo ""
echo "Phase 3: Symlinks"
echo "──────────────────"

declare -A SYMLINKS=(
  ["$COPILOT_DIR/copilot-instructions.md"]="$DOTFILES_DIR/copilot-instructions.md"
  ["$COPILOT_DIR/skills"]="$DOTFILES_DIR/skills"
  ["$COPILOT_DIR/agents"]="$DOTFILES_DIR/agents"
)

for target in "${!SYMLINKS[@]}"; do
  src="${SYMLINKS[$target]}"
  if [ -L "$target" ] && [ "$(readlink "$target")" = "$src" ]; then
    log_ok "$(basename "$target") → $src"
  elif [ -e "$target" ] && [ ! -L "$target" ]; then
    log_warn "$(basename "$target") exists but is not a symlink"
    if prompt_fix "Replace with symlink to $src"; then
      rm -rf "$target"
      ln -s "$src" "$target"
      log_fix "Replaced with symlink"
    fi
  else
    log_warn "$(basename "$target") not linked"
    if prompt_fix "Create symlink to $src"; then
      ln -s "$src" "$target"
      log_fix "Created symlink"
    fi
  fi
done

# ─── Phase 4: Config Merge ──────────────────────────────────────────────────
echo ""
echo "Phase 4: Config"
echo "────────────────"

if [ -f "$COPILOT_DIR/config.json" ]; then
  log_ok "config.json exists (not overwriting — merge manually if needed)"
  echo "         Source: $DOTFILES_DIR/config.json"
else
  log_warn "config.json missing"
  if prompt_fix "Copy config.json to ~/.copilot/"; then
    cp "$DOTFILES_DIR/config.json" "$COPILOT_DIR/config.json"
    log_fix "Copied config.json"
  fi
fi

# ─── Phase 5: MCP Registration ──────────────────────────────────────────────
echo ""
echo "Phase 5: MCP Servers"
echo "─────────────────────"

if [ -f "$MCP_CONFIG" ]; then
  if grep -q "session-context" "$MCP_CONFIG"; then
    log_ok "session-context registered in mcp-config.json"
  else
    log_warn "session-context not in mcp-config.json"
    if prompt_fix "Merge MCP config"; then
      # Simple merge — copy if empty, otherwise warn
      cp "$DOTFILES_DIR/mcp-config.json" "$MCP_CONFIG"
      log_fix "Copied mcp-config.json (review for manual merge if you had custom servers)"
    fi
  fi
else
  log_warn "mcp-config.json missing"
  if prompt_fix "Create mcp-config.json"; then
    cp "$DOTFILES_DIR/mcp-config.json" "$MCP_CONFIG"
    log_fix "Created mcp-config.json"
  fi
fi

# ─── Phase 6: Plugins ───────────────────────────────────────────────────────
echo ""
echo "Phase 6: Plugins"
echo "──────────────────"

if command -v copilot &>/dev/null; then
    {{PLUGIN_INSTALL_COMMANDS}}
    log_ok "Plugin installation complete"
else
    log_warn "copilot CLI not found — skipping plugin installation"
fi

# ─── Phase 7: Machine Identity ──────────────────────────────────────────────
echo ""
echo "Phase 7: Machine Identity"
echo "──────────────────────────"

if [ -f "$HOME/.machine-id" ]; then
  log_ok "~/.machine-id: $(cat "$HOME/.machine-id")"
else
  log_warn "~/.machine-id not found (shared with claude-dotfiles)"
  echo "         Create it: echo 'machine-name' > ~/.machine-id"
fi

# ─── Phase 8: MCP Server Build ──────────────────────────────────────────────
echo ""
echo "Phase 8: MCP Server Build"
echo "──────────────────────────"

MCP_DIR="$DOTFILES_DIR/mcp/session-context"
if [ -d "$MCP_DIR/dist" ]; then
  log_ok "session-context MCP server built"
else
  log_warn "session-context MCP server not built"
  if prompt_fix "Build MCP server (npm install && npm run build)"; then
    (cd "$MCP_DIR" && npm install && npm run build)
    log_fix "MCP server built"
  fi
fi

# ─── Summary ─────────────────────────────────────────────────────────────────
echo ""
echo "════════════════════════════"
echo -e "  Checks: $CHECKS  Passed: $PASSED  Fixed: $FIXED  Failed: $FAILED"
echo "════════════════════════════"

if [ "$FAILED" -gt 0 ]; then
  echo -e "${RED}Some checks failed. Fix the issues above and re-run.${NC}"
  exit 1
else
  echo -e "${GREEN}All checks passed.${NC}"
fi
```
